# 动态规划例子

## 斐波那契数列

```js
 *
 * @param {number} n
 * @returns {number}
 */
function fib(n) {
  if (n === 1 || n === 2) return 1;
  let pre1 = 1;
  let pre2 = 1;
  let result = 0;
  for (let i = 2; i < n; i++) {
    result = pre1 + pre2;
    pre2 = pre1;
    pre1 = result;
  }
  return result;
}
```

## 爬楼梯

```js
/**
 * dp(i) = dp(i - 1) + dp(i -2)
 * dp(1) = 1
 * dp(2) = 2
 * dp(3) = dp(1) + dp(2)
 * @param {number} n
 * @returns {number}
 */
function climbStairs(n) {
  if (n === 1 || n === 2) return n;
  let pre1 = 2;
  let pre2 = 1;
  let result = 0;
  for (let i = 2; i < n; i++) {
    result = pre1 + pre2;
    pre2 = pre1;
    pre1 = result;
  }
  return result;
}
```
